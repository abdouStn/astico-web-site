--vider le cache de symfony
php bin/console cache:clear
php bin/console cache:clear --env=prod

--generer un Bundle
php bin/console generate:bundle
	exple OC/PlatformBundle
-- utiliser composer pour installer un bundle:
php composer.phar --version
php composer.phar self-update

php ../composer.phar update  (se mettre dans le meme dossier que composer.json)
php composer.phar update friendsofsymfony/user-bundle

-- generer une entité
php bin/console generate:doctrine:entity

-- creer bdd symfony (une fois)
php bin/console doctrine:database:create

-- voir les requetes qu'il faut
php bin/console doctrine:schema:update --dump-sql

-- appliquer les requetes effectives
php bin/console doctrine:schema:update --force

-- generer les getters et setters automatiquement
php bin/console generate:doctrine:entities OCPlatformBundle:Advert

--remplir la bdd avec fixtures (dans src/DataFixtures/ORM/LoadXxx.php )  (ceci vide d'abord la bdd)
php bin/console doctrine:fixtures:load

--remplir la bdd avec FOSUserBundle:
php bin/console fos:user:create

--tester des requetes DQL
php bin/console doctrine:query:dql "SELECT a FROM OCPlatformBundle:Advert a"

--generer le dossier form pour les formulaires reutilisables ainsi que le fichier XxxType
php bin/console doctrine:generate:form OCPlatformBundle:Xxx

--liste des routes gerées pas FOSUserBundle
php bin/console router:debug 

--connaître tous les services implémentant un certain tag
php bin/console debug:container --tag=twig.extension

--Extraire les chaînes sources d'un site existant (pour traductions )
php bin/console translation:update --dump-messages --force fr OCPlatformBundle

--Exporter ses fichoers css et js avec Assetic
php bin/console assetic:dump --env=prod
php bin/console assets:install web

--Tester la vulnerabilité des dependances de son site
bin/console security:check

-- Rendre ses images, css, ... accessible pour web (copie dans web/bundles/X)
php bin/console assets:install web --symlink
